git branch -a

# SWITCHING BRANCHES 
git checkout other-branch-name

- git branch -m new-name

- rename unactive branch 
- git branch -m feature-one renamed-feature-one

# PUBLISHING A NEW BRANCH 
git push -u origin renamed-feature-one
